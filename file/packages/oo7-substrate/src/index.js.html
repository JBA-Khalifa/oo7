function tally(x) {
	var r = [0, 0]
	x.forEach(v => r[v ? 1 : 0]++)
	return {aye: r[1], nay: r[0]}
}

function tallyAmounts(x) {
	var r = [0, 0]
	x.forEach(([v, b]) => r[v ? 1 : 0] += b)
	return {aye: r[1], nay: r[0]}
}
